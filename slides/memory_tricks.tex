\section{Memory-Efficient Training Techniques}
\label{sec:memory_tricks}

% -----------------------------------------------------------------------
\begin{frame}{Motivation: The Memory Wall}

  \textbf{Recall:} Training with Adam requires $\sim 16\Psi$ bytes of model state \emph{plus} activations.
  Fine-tuning a 7B model naively needs $\sim$112\,GB, more than any single GPU.

  \vspace{0.3em}
  On modern GPUs, \textbf{memory is the bottleneck, not compute}:
  \begin{itemize}
    \item An H100 delivers 990\,TFLOPS (BF16) but only has 80\,GB HBM
    \item Compute is abundant; memory is the hard constraint
    \item It is therefore worthwhile to \emph{trade extra compute for reduced memory}
  \end{itemize}

  \vspace{0.5em}
  \textbf{Key question:} How can we reduce memory usage?

  \vspace{0.3em}
  \begin{description}
    \item[Reduce activation memory] \emphdef{Gradient checkpointing}: discard intermediate activations during forward pass, recompute them during backward. Trades $\sim$33\% extra compute for $O(\sqrt{L})$ instead of $O(L)$ activation memory.
    \item[Reduce parameter memory] \emphdef{Parameter-Efficient Fine-Tuning (PEFT)}: freeze the pretrained weights, only train a small number of additional parameters ($\sim$0.1--1\% of $\Psi$). Gradients and optimizer states are only needed for the small trainable set $\Rightarrow$ massive memory savings with minimal quality loss.
  \end{description}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{Gradient Checkpointing (Activation Recomputation)}

  \textbf{Problem:} Standard backpropagation stores all intermediate activations. Memory grows linearly with depth.

  \vspace{0.3em}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \textbf{Idea}~\cite{chen2016checkpointing}:
      \begin{itemize}
        \item During forward pass, only save activations at selected \emphdef{checkpoint} layers
        \item Discard all other intermediate activations
        \item During backward pass, \emph{recompute} missing activations from the nearest checkpoint
      \end{itemize}

      \vspace{0.3em}
      \textbf{Trade-off:}
      \begin{itemize}
        \item Memory: $O(\sqrt{L})$ instead of $O(L)$ for $L$ layers (with $\sqrt{L}$ evenly spaced checkpoints)
        \item Compute: $\sim$33\% extra forward-pass compute
      \end{itemize}

      \vspace{0.3em}
      \textbf{Selective recomputation}~\cite{korthikanti2023selective}: Only recompute attention (most memory-intensive). Saves $\sim$70\% activation memory at only $\sim$2.7\% compute overhead.
    \end{column}

\begin{frame}{Parameter-Efficient Fine-Tuning (PEFT): Overview}

  Full fine-tuning updates \emph{all} parameters. For a 7B model, this means storing 7B gradients + optimizer states.

  \vspace{0.3em}
  \textbf{PEFT idea:} Freeze the pretrained weights, only train a small number of \emph{additional} or \emph{selected} parameters.

  \vspace{0.5em}
  \begin{center}
  \begin{tabular}{l l l}
    \toprule
    \textbf{Method} & \textbf{What is trained} & \textbf{Trainable params} \\
    \midrule
    Full fine-tuning                    & All weights                      & 100\% \\
    LoRA~\cite{hu2022lora}              & Low-rank additive updates        & $\sim$0.1--1\% \\
    Adapters~\cite{houlsby2019adapters} & Small bottleneck modules         & $\sim$1--5\% \\
    Prefix Tuning~\cite{li2021prefix}   & Learnable prefix tokens          & $\sim$0.1\% \\
    Prompt Tuning~\cite{lester2021prompt} & Soft prompt embeddings         & $<$0.1\% \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.3em}
  All PEFT methods: gradients and optimizer states only for the small trainable set $\Rightarrow$ massive memory savings.
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{LoRA: Low-Rank Adaptation}

  \textbf{Key insight}~\cite{hu2022lora}: Weight updates during fine-tuning have low intrinsic rank.

  \vspace{0.3em}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      Instead of updating $W \in \mathbb{R}^{d \times k}$ directly, decompose the update:
      \[
        W' = W + \Delta W = W + BA
      \]
      where $B \in \mathbb{R}^{d \times r}$, $A \in \mathbb{R}^{r \times k}$, and $r \ll \min(d,k)$.

      \vspace{0.3em}
      \textbf{Training:}
      \begin{itemize}
        \item Freeze $W$ (no gradients, no optimizer state)
        \item Only train $A$ and $B$ ($r(d+k)$ params instead of $dk$)
        \item Initialize: $A \sim \mathcal{N}(0, \sigma^2)$, $B = 0$ (so $\Delta W = 0$ at start)
        \item Scale by $\frac{\alpha}{r}$ (hyperparameter $\alpha$ controls update magnitude)
      \end{itemize}

      \vspace{0.3em}
      \textbf{Inference:} Merge $W' = W + BA$: \emph{no extra latency}.
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering
      \begin{tikzpicture}[
        mat/.style={draw, minimum width=0.6cm, font=\scriptsize, inner sep=2pt, align=center},
      ]
        % Original weight
        \node[mat, fill=blue!15, minimum height=2.4cm, minimum width=1.8cm] (W) at (0, 0) {$W$\\{\tiny $d \times k$}\\{\tiny frozen}};

        % Plus sign
        \node[font=\large] at (1.5, 0) {$+$};

        % B matrix
        \node[mat, fill=orange!25, minimum height=2.4cm, minimum width=0.5cm] (B) at (2.5, 0) {$B$\\{\tiny $d \times r$}};

        % A matrix
        \node[mat, fill=green!25, minimum height=0.5cm, minimum width=1.8cm] (A) at (3.8, 0) {$A$ {\tiny $r \times k$}};

        % Brace for rank
        \draw[decorate, decoration={brace, amplitude=3pt, raise=2pt}]
          (B.north east) -- (B.south east)
          node[midway, right=6pt, font=\tiny] {$r \ll d$};

        % Labels
        \node[font=\scriptsize, gray] at (0, -1.7) {full rank $dk$};
        \node[font=\scriptsize, orange!70!black] at (3.2, -1.7) {low rank $r(d+k)$};
      \end{tikzpicture}

      \vspace{0.5em}
      \textbf{Example:} $d = k = 4096$, $r = 16$\\
      Full: 16.8M params per matrix\\
      LoRA: 131K params ($\sim$0.8\%)
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{LoRA: Practical Details}

  \begin{columns}
    \begin{column}{0.50\textwidth}
      \textbf{Which layers?}
      \begin{itemize}
        \item Typically applied to attention projections ($W_Q, W_K, W_V, W_O$)
        \item Can also apply to MLP layers
        \item More layers with smaller $r$ often beats fewer layers with larger $r$
      \end{itemize}

      \vspace{0.3em}
      \textbf{Typical hyperparameters:}
      \begin{itemize}
        \item Rank $r \in \{4, 8, 16, 32, 64\}$
        \item $\alpha$ often set to $2r$ or tuned
        \item Learning rate $\sim$3--10$\times$ higher than full fine-tuning
      \end{itemize}

      \vspace{0.3em}
      \textbf{Memory savings:}
      \begin{itemize}
        \item No gradient/optimizer for frozen weights
        \item 7B model: full FT needs $\sim$112\,GB, LoRA $\sim$16--18\,GB
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \textbf{Serving multiple LoRAs:}
      \begin{itemize}
        \item One base model in memory
        \item Swap/add small LoRA adapters per task or user
        \item Enables multi-tenant serving
      \end{itemize}

\end{frame}

\begin{frame}{Adapters}

  \textbf{Idea}~\cite{houlsby2019adapters}: Insert small trainable \emphdef{bottleneck modules} between existing layers.

  \vspace{0.3em}
  \begin{columns}
    \begin{column}{0.50\textwidth}
      \textbf{Architecture:}
      \begin{itemize}
        \item Project down: $h \mapsto \mathbb{R}^{m}$ where $m \ll h$
        \item Nonlinearity (e.g.\ ReLU/GELU)
        \item Project up: $\mathbb{R}^{m} \mapsto \mathbb{R}^{h}$
        \item Residual connection around the adapter
      \end{itemize}
      \[
        \text{Adapter}(x) = x + f(x W_{\text{down}}) W_{\text{up}}
      \]

      \vspace{0.3em}
      \textbf{Placement:} After attention and/or after FFN sublayer, before the residual add.

      \vspace{0.3em}
      \textbf{Comparison to LoRA:}
      \begin{itemize}
        \item Adapters add \emph{sequential} computation $\Rightarrow$ small inference overhead
        \item LoRA can be merged into weights $\Rightarrow$ zero overhead
        \item Both achieve similar quality at $\sim$1\% params
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \centering

\begin{frame}{Prefix Tuning and Prompt Tuning}

  \begin{columns}
    \begin{column}{0.50\textwidth}
      \textbf{Prefix Tuning}~\cite{li2021prefix}:
      \begin{itemize}
        \item Prepend learnable \emphdef{prefix vectors} to keys and values in every attention layer
        \item Prefix length $l$ (e.g.\ 10--200 tokens)
        \item Parameters: $l \times L \times 2 \times d$ (prefix per layer, for $K$ and $V$)
        \item Typically $<$0.1\% of model parameters
      \end{itemize}

      \vspace{0.5em}
      \textbf{Prompt Tuning}~\cite{lester2021prompt}:
      \begin{itemize}
        \item Even simpler: learnable \emphdef{soft tokens} prepended at the \emph{input embedding} level only
        \item No changes to internal layers
        \item Parameters: $l \times d$ (one embedding per soft token)
        \item Scales well: approaches full FT quality for large models ($>$10B)
      \end{itemize}
    \end{column}

\begin{frame}{Summary and Comparison}

  \begin{center}
  \renewcommand{\arraystretch}{1.2}
  \begin{tabular}{l c c c c}
    \toprule
    \textbf{Method} & \textbf{Memory} & \textbf{Trainable} & \textbf{Inference} & \textbf{Quality} \\
    & \textbf{saving} & \textbf{params} & \textbf{overhead} & \\
    \midrule
    Grad.\ checkpointing  & Activations      & 100\%       & None       & Same \\
    LoRA                   & Large            & $\sim$0.5\% & None (merged) & Near full FT \\
    Adapters               & Large            & $\sim$2\%   & Small      & Near full FT \\
    Prefix Tuning          & Very large       & $<$0.1\%    & Prefix KV  & Good ($\geq$10B) \\
    Prompt Tuning          & Very large       & $<$0.1\%    & Extra tokens & Good ($\geq$10B) \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.3em}
  \textbf{In practice:}
  \begin{itemize}
    \item \textbf{Pretraining:} Gradient checkpointing + mixed precision + parallelism (always)
    \item \textbf{Fine-tuning on limited hardware:} LoRA (most popular PEFT method)
    \item \textbf{Multi-task serving:} LoRA adapters swapped per task on shared base model
  \end{itemize}
\end{frame}
