\begin{frame}{First Moment: Smoothing the Gradient with Momentum}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Problem:} stochastic gradients oscillate, especially in directions of high curvature.

      \vspace{0.4em}
      \textbf{Idea:} maintain a running average of gradients to smooth out noise.

      \vspace{0.4em}
      \textbf{First moment} (exponential moving average of $\nabla_W L$):
      \[
        \underbrace{M_t}_{\text{new}} = \beta_1 \underbrace{M_{t-1}}_{\text{prev.}} + (1-\beta_1) \underbrace{\nabla_W L}_{\text{gradient}}
      \]
      typically $\beta_1 = 0.9$.

      \vspace{0.4em}
      \textbf{Update with $M_t$ in place of the raw gradient:}
      \[
        W_t = W_{t-1} - \eta \cdot M_t
      \]

      \vspace{0.4em}
      {\small Oscillations in the high-curvature direction cancel; the consistent signal in the low-curvature direction accumulates $\Rightarrow$ smoother and faster descent.}
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{center}
      \textbf{\small Computing $M_t$ from $M_{t-1}$ and $\nabla L$:}
      \vspace{0.2em}

      \begin{tikzpicture}[scale=0.85, >=stealth]
        \fill[black] (0, 0) circle (1.5pt);

        % β * M_{t-1}: long horizontal blue arrow (large weight)
        \draw[->, very thick, blue!70!black] (0, 0) -- (2.6, 0);
        \node[font=\scriptsize, blue!70!black, anchor=south] at (1.3, 0.05) {$\beta\, M_{t-1}$};

        % (1-β) * ∇L: small red arrow pointing down (small weight)

\begin{frame}{Second Moment: Per-Parameter Adaptive Learning Rate}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Problem:} different parameters have wildly different gradient magnitudes; a single global learning rate cannot fit all.

      \vspace{0.4em}
      \textbf{Idea:} track the EMA of \emph{squared} gradients per parameter to estimate gradient scale.

      \vspace{0.4em}
      \textbf{Second moment} (EMA of $(\nabla_W L)^2$, element-wise):
      \[
        \underbrace{V_t}_{\text{new}} = \beta_2 \underbrace{V_{t-1}}_{\text{prev.}} + (1-\beta_2) \underbrace{(\nabla_W L)^2}_{\text{sq.\ gradient}}
      \]
      typically $\beta_2 = 0.999$.

      \vspace{0.4em}
      \textbf{Adaptive update:}
      \[
        W_t = W_{t-1} - \eta \cdot \frac{M_t}{\sqrt{V_t} + \epsilon}
      \]

      \vspace{0.4em}
      {\small Each parameter gets its own effective learning rate based on its gradient history.}
    \end{column}

\begin{frame}{Adam's Per-Parameter Adaptivity}

  Adam normalizes each gradient coordinate independently:
  \[
    \Delta w_i \;\propto\; \frac{\hat{m}_i}{\sqrt{\hat{v}_i} + \epsilon}
    \qquad \Rightarrow \qquad |\Delta w_i| \approx \pm \eta
  \]

  \vspace{0.3em}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \textbf{Advantages:}
      \begin{itemize}
        \item Parameters with sparse/small gradients still get meaningful updates
        \item Relatively insensitive to gradient scale $\Rightarrow$ less LR tuning
        \item Works well across embeddings, attention, MLP
      \end{itemize}

      \vspace{0.3em}
      \textbf{Limitation for matrix parameters:}
      \begin{itemize}
        \item Adam treats each entry $W_{ij}$ independently
        \item Ignores the \emph{matrix structure} of $G = \nabla_W L$
        \item Principal directions of $G$ (singular vectors $u_i$) are generally \emph{not aligned} with coordinate axes
        \item Element-wise rescaling can stretch along axes but cannot ``un-tilt'' the geometry
      \end{itemize}

      \vspace{0.3em}
      $\Rightarrow$ Can we design an optimizer that respects the matrix structure of 2D parameters?
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{center}
      \textbf{The limitation:}
      \vspace{0.3em}

      \begin{tikzpicture}[scale=0.65, >=stealth]

\begin{frame}{AdamW: Decoupled Weight Decay~\cite{loshchilov2019adamw}}

  \textbf{Weight decay} regularizes by penalizing large weights:
  \[
    L_{\text{reg}}(W) = L(W) + \tfrac{\lambda}{2} \|W\|^2
  \]

  \vspace{0.3em}
  \textbf{Coupled (Adam + L2):} add $\lambda W_{t-1}$ to the gradient $G_t$ before the adaptive update.
  \begin{itemize}
    \item Decay magnitude is then divided by $\sqrt{\hat{V}_t} + \epsilon$
    \item Effective decay strength differs per parameter $\Rightarrow$ inconsistent regularization
  \end{itemize}

  \vspace{0.3em}
  \textbf{Decoupled (AdamW):} apply weight decay \emph{directly} to the parameters, separate from the adaptive step:
  \[
    W_t = W_{t-1} - \eta \left( \frac{\hat{M}_t}{\sqrt{\hat{V}_t} + \epsilon} + \lambda W_{t-1} \right)
  \]

  \vspace{0.3em}
  \begin{itemize}
    \item Decay applied uniformly, independent of the adaptive update
    \item Improves generalization $\Rightarrow$ standard for LLM training
  \end{itemize}

\end{frame}

\begin{frame}{Bias Correction: Why $\hat{M}_t$ and $\hat{V}_t$?}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Issue:} moments are initialized at zero ($M_0 = V_0 = 0$).

      \vspace{0.3em}
      For a stationary gradient with mean $\mu$:
      \[
        \mathbb{E}[M_t] = (1-\beta_1^t)\,\mu
      \]
      $M_t$ is biased \emph{toward zero} early in training.

      \vspace{0.3em}
      Example with $\beta_1 = 0.9$:
      \begin{itemize}
        \item $t=1$: $\mathbb{E}[M_1] = 0.1\,\mu$ (only 10\% of true)
        \item $t=10$: $\mathbb{E}[M_{10}] \approx 0.65\,\mu$
        \item $t=100$: $\mathbb{E}[M_{100}] \approx \mu$
      \end{itemize}

      \vspace{0.3em}
      \textbf{Fix: divide by the bias factor.}
      \[
        \hat{M}_t = \frac{M_t}{1-\beta_1^t}, \quad \hat{V}_t = \frac{V_t}{1-\beta_2^t}
      \]

      Now $\mathbb{E}[\hat{M}_t] = \mu$ at every step.
    \end{column}
    \begin{column}{0.47\textwidth}
      \begin{center}
      \textbf{$M_t/\mu$ vs step (with/without correction)}
      \vspace{0.3em}

      \begin{tikzpicture}[scale=0.65, >=stealth]
        \draw[->, thick] (0,0) -- (5.2,0) node[right, font=\scriptsize] {step $t$};
        \draw[->, thick] (0,0) -- (0,3.4);

        % Reference line at 1 (true value μ)
        \draw[gray, dashed] (0, 2.7) -- (5, 2.7);
        \node[font=\tiny, gray, anchor=east] at (-0.05, 2.7) {$\mu$};

        % Without correction
        \draw[red!70!black, very thick, smooth] plot coordinates {
          (0, 0) (0.5, 1.05) (1.0, 1.7) (1.5, 2.1) (2.0, 2.35) (2.5, 2.5)
          (3.0, 2.59) (3.5, 2.64) (4.0, 2.67) (4.5, 2.69) (4.8, 2.7)
        };
        \node[font=\scriptsize, red!70!black, anchor=west] at (3.2, 2.0) {$M_t$};

        % With correction (= μ from start)
        \draw[blue!70!black, very thick] (0.05, 2.7) -- (4.9, 2.7);
        \node[font=\scriptsize, blue!70!black, anchor=south west] at (1.5, 2.75) {$\hat{M}_t$};
      \end{tikzpicture}

      \vspace{0.4em}
      \textbf{Even worse for $V$:}

      \vspace{0.2em}
      With $\beta_2 = 0.999$, at $t=1$, $V_1$ captures only $\approx 0.1\%$ of the true variance. Without correction, $\sqrt{V_1}$ in the denominator would explode the update.

      \vspace{0.2em}
      $\hat{V}_1 = V_1 / 0.001$ recovers the right scale.
      \end{center}
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{Adam / AdamW: Adaptive Moment Estimation~\cite{kingma2015adam, loshchilov2019adamw}}

  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{algorithm}[H]
        \small
        \caption{Adam / AdamW}
        \KwInput{LR $\eta$, decay rates $\beta_1, \beta_2$, $\epsilon$; AdamW: weight decay $\lambda$}
        Initialize $M_0 \gets 0, V_0 \gets 0$\;
        \For{$t = 1, 2, \ldots$}{
          $G_t \gets \nabla_W L(W_{t-1})$\;
          $M_t \gets \beta_1 M_{t-1} + (1{-}\beta_1) G_t$\;
          $V_t \gets \beta_2 V_{t-1} + (1{-}\beta_2) G_t^2$\;
          $\hat{M}_t \gets M_t / (1{-}\beta_1^t)$\;
          $\hat{V}_t \gets V_t / (1{-}\beta_2^t)$\;
          {\color{blue!70!black} $W_t \gets W_{t-1} - \eta\, \hat{M}_t / (\sqrt{\hat{V}_t} + \epsilon)$}\tcp*{\color{blue!70!black} Adam}
          {\color{slide_violet} $W_t \gets W_{t-1} - \eta\, [\hat{M}_t / (\sqrt{\hat{V}_t} + \epsilon) + \lambda W_{t-1}]$}\tcp*{\color{slide_violet} AdamW}
        }
      \end{algorithm}
    \end{column}
    \begin{column}{0.42\textwidth}
      \textbf{Components:}
      \begin{itemize}
        \item $M_t$: momentum-smoothed gradient
        \item $V_t$: tracks per-parameter gradient magnitude
        \item $\hat{M}_t, \hat{V}_t$: bias-corrected moments
        \item \textcolor{slide_violet}{$\lambda W_{t-1}$}: AdamW's decoupled weight decay
      \end{itemize}

      \vspace{0.3em}
      \textbf{Defaults:} $\beta_1 {=} 0.9$, $\beta_2 {=} 0.999$, $\epsilon {=} 10^{-8}$\\
      \textbf{LLM defaults:} $\beta_2 {=} 0.95$, AdamW with $\lambda {\approx} 0.1$

      \vspace{0.4em}
      \begin{alertblock}{Critical: warmup}
        Early-training $\hat{V}_t$ from few samples is unreliable $\Rightarrow$ huge updates. \textbf{Always linearly ramp $\eta$ from $0$ to $\eta_{\max}$ over $\sim$100--2000 steps.} Same applies to Muon.
      \end{alertblock}
    \end{column}
  \end{columns}

\end{frame}

\subsection{The Muon Optimizer}

% -----------------------------------------------------------------------
\begin{frame}{Muon: Momentum + Orthogonalization~\cite{jordan2024muon}}

  \textbf{Key insight:} For 2D weight matrices, \emph{orthogonalize} the momentum buffer before applying the update. This ensures all singular directions are updated equally.

  \vspace{0.4em}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Algorithm overview:}
      \begin{enumerate}
        \item Compute gradient $G_t = \nabla_W L$
        \item Accumulate with \textbf{Nesterov momentum}:
        \[
          M_t = \beta \cdot M_{t-1} + G_t
        \]
        \item \textbf{Orthogonalize} the momentum via Newton-Schulz iteration:
        \[
          U_t = \text{NS}_5(M_t) \approx \text{Ortho}(M_t)
        \]
        \item Update: $W_t = W_{t-1} - \eta \cdot U_t$
      \end{enumerate}

      \vspace{0.3em}
      {\small Apply Muon only to 2D (matrix) parameters.\\
      Use AdamW for embeddings, classifier head, biases, and all 1D parameters.}
    \end{column}
    \begin{column}{0.47\textwidth}
      \textbf{What does orthogonalization do?}

      Given $G = U \Sigma V^\top$ (SVD), the nearest semi-orthogonal matrix is:
      \[
        \text{Ortho}(G) = U V^\top
      \]

      This \emph{replaces all singular values with 1}:

      \vspace{0.2em}
      \begin{center}

\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{Why Orthogonalize? The Condition Number Problem}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Observation:} Gradient matrices for 2D transformer parameters typically have \emph{very high condition numbers}.

      \vspace{0.3em}
      A few singular directions dominate the update, while most directions receive negligible signal.

      \vspace{0.3em}
      \textbf{SGD/Adam behavior:}
      \begin{itemize}
        \item SGD: all neuron updates dominated by the top few singular vectors
        \item Adam: element-wise rescaling partially helps, but does not fix the directional imbalance
      \end{itemize}

      \vspace{0.3em}
      \textbf{Orthogonalization:}
      \begin{itemize}
        \item Sets all singular values to 1
        \item Every direction in the update gets equal weight
        \item ``Amplifies rare directions'' that have small magnitude but are important for learning
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}
        \begin{tikzpicture}[scale=0.7]
          % Singular value spectrum
          \draw[->, thick] (0,0) -- (5.5,0) node[right, font=\scriptsize] {index $i$};
          \draw[->, thick] (0,0) -- (0,4) node[above, font=\scriptsize] {$\sigma_i$};
          \node[font=\scriptsize\bfseries] at (2.5,4.4) {Singular Value Spectrum of $\nabla_W L$};

          % Typical gradient spectrum - steep decay
          \draw[red!70!black, very thick, smooth] plot coordinates {
            (0.3, 3.5) (0.8, 2.4) (1.3, 1.6) (1.8, 1.0) (2.3, 0.6)
            (2.8, 0.35) (3.3, 0.2) (3.8, 0.12) (4.3, 0.07) (4.8, 0.04)
          };
          \node[font=\scriptsize, red!70!black, anchor=west] at (3.5, 3.0) {typical gradient};

          % After orthogonalization - flat
          \draw[green!50!black, very thick, dashed] (0.3, 1.5) -- (4.8, 1.5);
          \node[font=\scriptsize, green!50!black, anchor=west] at (3.5, 1.8) {after Ortho};

          % Annotation
          \draw[<->, gray, thin] (4.8, 0.04) -- (4.8, 1.5);
          \node[font=\tiny, gray, anchor=west] at (5.0, 0.75) {amplified};

      \end{center}

      \vspace{0.3em}
      \textbf{Connection to steepest descent:}

      Orthogonalized gradient $UV^\top$ is the \emph{steepest descent direction} under the \textbf{spectral norm}~\cite{bernstein2024oldoptimizer}:
      \[
        UV^\top = \arg\min_{\|D\|_\text{op} \leq 1} \langle G, D \rangle
      \]
    \end{column}
  \end{columns}
\end{frame}

\subsection{Newton-Schulz Iteration}

% -----------------------------------------------------------------------
\begin{frame}{Newton-Schulz Iteration: Idea}

  \textbf{Goal:} compute $\text{Ortho}(X) = U V^\top$ without explicit SVD (slow, not GPU-friendly).

  \vspace{0.4em}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Key fact:} an odd polynomial in $X$ acts on the \emph{singular values only}, leaving singular vectors untouched:
      \[
        X = U \Sigma V^\top \;\;\Longrightarrow\;\; p(X) = U \, p(\Sigma) \, V^\top
      \]

      \vspace{0.3em}
      \textbf{Strategy:}
      \begin{itemize}
        \item Pick a polynomial $p$ that pulls every $\sigma \in (0, 1]$ toward $1$
        \item Apply repeatedly: $X_{k+1} = p(X_k)$
        \item After a few iterations all $\sigma_i \approx 1$ $\Rightarrow$ $X_K \approx UV^\top$
      \end{itemize}

      \vspace{0.3em}
      \textbf{Why this is great:}
      \begin{itemize}
        \item Only \textbf{matrix multiplications} -- ideal for GPUs
        \item No SVD, no eigendecomposition, no division
        \item Geometric convergence: $\sim 5$ steps suffice
        \item We never explicitly compute $U, \Sigma, V$
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{center}
        \textbf{Iterating $p$ on a singular value $\sigma$}
        \vspace{0.2em}

        \begin{tikzpicture}[scale=0.7, >=stealth]

          \draw[->, thick] (0,0) -- (3.4,0) node[right, font=\scriptsize] {$\sigma$};

\begin{frame}{Newton-Schulz Implementation}

  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \begin{algorithm}[H]
        \small
        \caption{Newton-Schulz Orthogonalization}
        \KwInput{Gradient $G \in \mathbb{R}^{n \times m}$, steps $T{=}5$}
        $(a, b, c) \gets (3.4445,\; -4.7750,\; 2.0315)$\;
        $X \gets G / \|G\|_F$ \tcp*{$\sigma_i \in [0,1]$}
        \If{$n > m$}{$X \gets X^\top$}
        \For{$t = 1, \ldots, T$}{
          $A \gets X X^\top$\;
          $B \gets b \cdot A + c \cdot A^2$\;
          $X \gets a \cdot X + BX$ \tcp*{quintic poly}
        }
        \If{$n > m$}{$X \gets X^\top$}
        \Return{$X \approx UV^\top$}
      \end{algorithm}

      \vspace{0.3em}
      {\small Runs entirely in \textbf{bfloat16}. If $n > m$, transpose before iterating to work on the smaller dimension.}
    \end{column}
    \begin{column}{0.47\textwidth}
      \textbf{Computational cost per NS step} (for $n {\times} m$, $m {\leq} n$):
      \begin{itemize}
        \item $XX^\top$: $2nm^2$ FLOPs
        \item $A \cdot A$: $2m^3$ FLOPs
        \item $B \cdot X$: $2nm^2$ FLOPs
        \item Total per step: $\leq 6nm^2$ FLOPs
        \item For 5 steps: $30nm^2$ FLOPs
      \end{itemize}

      \vspace{0.3em}
      \textbf{Relative overhead:}
      \begin{itemize}
        \item $\leq 5m/B$ of the forward pass ($B$ = batch size in tokens)
        \item Negligible for typical LLM training batch sizes ($B \gg m$)
      \end{itemize}
    \end{column}
  \end{columns}

\end{frame}

\begin{frame}{Full Muon Algorithm}

  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \begin{algorithm}[H]
        \small
        \caption{Muon Optimizer}
        \KwInput{LR $\eta$, momentum $\beta$, NS steps $T{=}5$}
        Initialize $M_0 = 0$ for each 2D param $W$\;
        \For{$t = 1, 2, \ldots$}{
          $G_t \gets \nabla_W L(W_{t-1})$\;
          \tcp{Nesterov momentum}
          $M_t \gets \beta \cdot M_{t-1} + G_t$\;
          $\tilde{M}_t \gets \beta \cdot M_t + G_t$\;
          \tcp{Orthogonalize via NS}
          $U_t \gets \text{NewtonSchulz}_T(\tilde{M}_t)$\;
          \tcp{Update weights}
          $W_t \gets W_{t-1} - \eta \cdot U_t$\;
        }
      \end{algorithm}

      \vspace{0.3em}
      \textbf{Practical details:}
      \begin{itemize}
        \item Muon: 2D weight matrices (Q, K, V, MLP)
        \item AdamW: embeddings, output head, biases, LayerNorm
        \item Apply to Q, K, V \emph{separately}, not concatenated
      \end{itemize}
    \end{column}

\begin{frame}{Muon: Summary and Practical Advice}

  \begin{block}{Key Takeaways}
    \begin{itemize}
      \item \textbf{Muon = Momentum + Orthogonalization} via Newton-Schulz iteration
      \item Orthogonalization ensures all singular directions get equal-magnitude updates
      \item Equivalent to steepest descent under the spectral norm
      \item Mathematically equivalent to instantaneous (accumulation-free) Shampoo
      \item Newton-Schulz iteration: 5 steps of matrix multiplications in bfloat16, negligible overhead
      \item 1.35$\times$ sample-efficiency improvement over AdamW on NanoGPT benchmark
    \end{itemize}
  \end{block}

  \vspace{0.3em}
  \textbf{When to use:}
  \begin{description}[leftmargin=0.3em, labelwidth=\widthof{\bfseries Muon + AdamW}]
    \item[Muon] All 2D weight matrices: Q, K, V projections, MLP layers (apply separately, not concatenated)
    \item[AdamW] Embeddings, output head, biases, LayerNorm/RMSNorm parameters, all 1D tensors
  \end{description}
