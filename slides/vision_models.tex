\begin{frame}[fragile]{ViT: Patch Embedding \& Transformer Block}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Patch embedding}:
      \[
        \mathbf{z}_0 = [\mathbf{x}_\text{class};\; \mathbf{x}_p^1 E;\; \ldots;\; \mathbf{x}_p^N E] + E_\text{pos}
      \]
      \begin{itemize}
        \item $E \in \mathbb{R}^{(P^2 C) \times D}$: projection matrix
        \item $E_\text{pos} \in \mathbb{R}^{(N+1) \times D}$: positional embeddings
        \item Implemented as \texttt{Conv2d} with kernel and stride $= P$
      \end{itemize}

      \vspace{0.5em}
      \textbf{Transformer encoder block} (pre-norm):
      \begin{align*}
        \mathbf{z}'_\ell &= \operatorname{MSA}(\operatorname{LN}(\mathbf{z}_{\ell-1})) + \mathbf{z}_{\ell-1} \\
        \mathbf{z}_\ell &= \operatorname{MLP}(\operatorname{LN}(\mathbf{z}'_\ell)) + \mathbf{z}'_\ell
      \end{align*}
      Classification: $\mathbf{y} = \operatorname{LN}(\mathbf{z}_L^0)$
    \end{column}
    \begin{column}{0.48\textwidth}

\begin{frame}{DeiT: Hard vs Soft Distillation}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Soft distillation} (KL divergence):
      \[
        \mathcal{L}_\text{soft} = (1{-}\lambda)\,\mathcal{L}_\text{CE}(\psi(Z_s), y) + \lambda\,\tau^2\, \text{KL}(\psi(Z_s/\tau) \| \psi(Z_t/\tau))
      \]
      \begin{itemize}
        \item $Z_s, Z_t$: student/teacher logits
        \item $\tau$: temperature, $\psi$: softmax
      \end{itemize}

      \vspace{0.5em}
      \textbf{Hard distillation} (best results):
      \[
        \mathcal{L}_\text{hard} = \tfrac{1}{2}\,\mathcal{L}_\text{CE}(\psi(Z_s), y) + \tfrac{1}{2}\,\mathcal{L}_\text{CE}(\psi(Z_s), y_t)
      \]
      where $y_t = \arg\max Z_t$ (teacher's hard prediction)
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Training strategy} (data-efficient):
      \begin{itemize}
        \item ImageNet-1K only (1.2M images)
        \item Strong augmentation: RandAugment, Mixup, CutMix, Random Erasing
        \item Regularization: Stochastic Depth, repeated augmentation
        \item AdamW + cosine schedule
        \item Trains in $<3$ days on 8 GPUs
      \end{itemize}

      \vspace{0.3em}
      \begin{center}
        \small
        \begin{tabular}{lcc}
          \toprule
          \textbf{Variant} & \textbf{Dim} & \textbf{Params} \\
          \midrule
          DeiT-Ti & 192 & 5M \\
          DeiT-S & 384 & 22M \\
          DeiT-B & 768 & 86M \\
          \bottomrule
        \end{tabular}
      \end{center}

      \vspace{0.3em}
      Key finding: CNN teacher (RegNet) outperforms Transformer teacher for distillation.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{MAE: Design Choices \& Results}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Why 75\% masking?}
      \begin{itemize}
        \item Images have high spatial redundancy --- low masking allows trivial interpolation
        \item 75\% forces the model to learn \emph{semantic} understanding, not local texture
        \item Much higher than BERT's 15\% for text
      \end{itemize}

      \vspace{0.3em}
      \textbf{Asymmetric encoder-decoder}:
      \begin{itemize}
        \item Encoder: large ViT (e.g.\ 24 layers, 1024 dim)
        \item Decoder: small (8 layers, 512 dim)
        \item Decoder \textbf{discarded} after pre-training
        \item Only encoder used for downstream tasks
      \end{itemize}

      \vspace{0.3em}
      \textbf{Downstream usage}:
      \begin{description}[style=nextline, leftmargin=0.5cm]
        \item[Fine-tuning] Full encoder + classification head; best results
        \item[Linear probing] Frozen encoder + linear layer; weaker than contrastive methods but gap narrows at scale
      \end{description}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{ImageNet-1K results} (fine-tuning):
      \begin{center}
        \small
        \begin{tabular}{lcc}
          \toprule
          \textbf{Model} & \textbf{Params} & \textbf{Top-1} \\
          \midrule
          ViT-B/16 & 86M & 83.6\% \\
          ViT-L/16 & 307M & 85.9\% \\
          ViT-H/14 & 632M & 86.9\% \\
          \bottomrule
        \end{tabular}
      \end{center}
      Using ImageNet-1K only --- no external data.

      \vspace{0.3em}
      \textbf{vs.\ other self-supervised methods}:
      \begin{center}
        \small
        \begin{tabular}{lcc}
          \toprule
           & \textbf{Fine-tune} & \textbf{Lin.\ probe} \\

      \end{center}

      \vspace{0.3em}
      \textbf{Key}: MAE scales excellently to ViT-H; contrastive methods plateau at larger sizes.
    \end{column}
  \end{columns}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{Swin: Window \& Shifted Window Attention}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Window-MSA (W-MSA)}:
      \begin{itemize}
        \item Partition feature map into non-overlapping $M{\times}M$ windows ($M{=}7$)
        \item Self-attention computed \emph{within} each window
        \item Complexity: $\mathcal{O}(M^2 \cdot hw \cdot C)$ --- \textbf{linear} in image size
      \end{itemize}

      \vspace{0.3em}
      \textbf{Problem}: No cross-window information flow.

      \vspace{0.3em}
      \textbf{Shifted Window-MSA (SW-MSA)}:
      \begin{itemize}
        \item Shift window grid by $(\lfloor M/2\rfloor, \lfloor M/2\rfloor)$
        \item Windows now straddle previous boundaries
        \item Consecutive blocks alternate W-MSA / SW-MSA
      \end{itemize}

      \vspace{0.3em}
      \textbf{Efficient cyclic shift}:
      \begin{enumerate}
        \item Cyclically shift feature map
        \item Regular window partition + attention masks
        \item Reverse shift
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{center}
        \begin{tikzpicture}[scale=0.65, every node/.style={scale=0.65}]
          % --- Layer l: regular windows ---

\begin{frame}{Swin: Complexity Comparison \& Configurations}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Swin block equations}:
      \begin{align*}
        \hat{\mathbf{z}}^\ell &= \text{W-MSA}(\text{LN}(\mathbf{z}^{\ell-1})) + \mathbf{z}^{\ell-1} \\
        \mathbf{z}^\ell &= \text{MLP}(\text{LN}(\hat{\mathbf{z}}^\ell)) + \hat{\mathbf{z}}^\ell \\
        \hat{\mathbf{z}}^{\ell+1} &= \text{SW-MSA}(\text{LN}(\mathbf{z}^\ell)) + \mathbf{z}^\ell \\
        \mathbf{z}^{\ell+1} &= \text{MLP}(\text{LN}(\hat{\mathbf{z}}^{\ell+1})) + \hat{\mathbf{z}}^{\ell+1}
      \end{align*}

      \vspace{0.3em}
      \textbf{Complexity} ($h{\times}w$ spatial, $M^2$ window):
      \begin{description}[leftmargin=0.5cm]
        \item[Global MSA] $\mathcal{O}(4hwC^2 + 2(hw)^2 C)$
        \item[W-MSA] $\mathcal{O}(4hwC^2 + 2M^2 hwC)$
      \end{description}
      Quadratic $\to$ \textbf{linear} in spatial resolution.
    \end{column}

      \end{center}

      \vspace{0.5em}
      \textbf{Key advantages}:
      \begin{itemize}
        \item Hierarchical features $\to$ natural fit as backbone for detection, segmentation

\begin{frame}{MLP-Mixer: Architecture Comparison}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Full pipeline}:
      \begin{enumerate}
        \item Split image into $S = HW/P^2$ patches
        \item Per-patch linear projection $\to X \in \mathbb{R}^{S \times C}$
        \item $N$ Mixer layers (token-mix + channel-mix)

      \vspace{0.3em}
      \textbf{Why it works}:
      \begin{itemize}
        \item Token-mixing MLPs learn \emph{fixed} spatial mixing patterns (no data-dependent attention)
        \item With enough data, fixed mixing is competitive with attention
        \item Much simpler architecture
      \end{itemize}
    \end{column}

          \midrule
          Spatial mix & Attention & Token MLP \\
          Channel mix & FFN & Channel MLP \\
          Data-dep.\ & Yes & No \\
          Complexity & $\mathcal{O}(S^2)$ & $\mathcal{O}(S)$ \\
          \bottomrule
        \end{tabular}
      \end{center}

      \textbf{Configurations}:
      \begin{center}

      \end{center}

      \vspace{0.3em}

  \end{columns}

\begin{frame}{DETR: Bipartite Matching Loss}

  Training requires matching $N$ predictions to $M$ ground truth objects (padded with $\varnothing$).

  \vspace{0.3em}
  \begin{columns}
    \begin{column}{0.52\textwidth}
      \textbf{Step 1: Hungarian matching} --- find optimal assignment $\hat{\sigma}$:
      \[
        \hat{\sigma} = \arg\min_{\sigma \in \mathfrak{S}_N} \sum_{i=1}^{N} \mathcal{L}_\text{match}(y_i, \hat{y}_{\sigma(i)})
      \]
      \[
        \mathcal{L}_\text{match} = -\mathbbm{1}_{c_i \neq \varnothing}\, \hat{p}_{\sigma(i)}(c_i) + \mathbbm{1}_{c_i \neq \varnothing}\, \mathcal{L}_\text{box}(b_i, \hat{b}_{\sigma(i)})
      \]
      Solved by the Hungarian algorithm in $\mathcal{O}(N^3)$.

      \vspace{0.3em}
      \textbf{Step 2: Compute loss} on matched pairs:
      \[
        \mathcal{L} = \sum_{i=1}^{N} \bigl[-\log \hat{p}_{\hat{\sigma}(i)}(c_i) + \mathbbm{1}_{c_i \neq \varnothing}\, \mathcal{L}_\text{box}\bigr]
      \]
      \[
        \mathcal{L}_\text{box} = \lambda_\text{iou}\,\mathcal{L}_\text{giou} + \lambda_{L_1}\,\| b_i - \hat{b} \|_1
      \]
    \end{column}
    \begin{column}{0.44\textwidth}
      \begin{center}
        \begin{tikzpicture}[
          scale=0.8, every node/.style={scale=0.8},
          pred/.style={draw, circle, fill=blue!15, inner sep=3pt, font=\scriptsize},
          gt/.style={draw, circle, fill=green!20, inner sep=3pt, font=\scriptsize},
        ]
          % Ground truth

  \vspace{0.2em}
  \textbf{Deformable attention}:
  \[
    \operatorname{DeformAttn}(\mathbf{z}_q, \mathbf{p}_q, \mathbf{x}) = \sum_{m=1}^{M} W_m \biggl[\sum_{k=1}^{K} A_{mqk} \cdot W'_m \, \mathbf{x}(\mathbf{p}_q + \Delta\mathbf{p}_{mqk})\biggr]
  \]
  \begin{itemize}
    \item Offsets $\Delta\mathbf{p}_{mqk}$ and weights $A_{mqk}$ predicted from query $\mathbf{z}_q$ via linear layers
    \item $\mathbf{x}(\cdot)$: bilinear interpolation at continuous positions
    \item $K{=}4$ sampling points per head, $M$ heads
  \end{itemize}
\end{frame}

\begin{frame}{Deformable DETR: Multi-Scale Features}

  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Multi-scale deformable attention}:
      \[
        \operatorname{MSDeformAttn} = \sum_{m=1}^{M} W_m \biggl[\sum_{l=1}^{L}\sum_{k=1}^{K} A_{mlqk} \cdot W'_m \, \mathbf{x}^l(\phi_l(\hat{\mathbf{p}}_q) + \Delta\mathbf{p}_{mlqk})\biggr]

      \begin{itemize}
        \item Attend across $L{=}4$ feature map levels simultaneously
        \item $\phi_l$: rescales reference point to level $l$

      \end{itemize}

      \vspace{0.3em}
      \textbf{Additional features}:
      \begin{itemize}
        \item \textbf{Iterative box refinement}: each decoder layer refines the box from the previous layer
        \item \textbf{Two-stage variant}: encoder produces proposals as initial reference points
