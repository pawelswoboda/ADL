\section{State Space Models and Mamba}

% =======================================================================
\subsection{State Space Models}

% -----------------------------------------------------------------------
\begin{frame}{Motivation: Recurrence vs.\ Attention}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{RNNs / LSTMs}
      \begin{itemize}
        \item $O(1)$ memory per step, streaming
        \item Sequential $\Rightarrow$ \alert{not parallelizable} during training
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Transformers}
      \begin{itemize}
        \item Fully parallel training
        \item \alert{$O(T^2)$} attention cost
        \item KV cache grows linearly at inference
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.8em}
  \begin{alertblock}{Can we get the best of both?}
    \textbf{State Space Models} aim to combine:
    \begin{itemize}
      \item Constant-size hidden state $\Rightarrow$ $O(1)$ inference per step (like RNNs)
      \item Parallel training via convolutional or attention-like view (like Transformers)
    \end{itemize}
  \end{alertblock}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{Continuous-Time State Space Models}

  SSMs originate from control theory. A continuous-time, linear system maps input $x(t) \in \R$ to output $y(t) \in \R$ via a latent state $h(t) \in \R^N$:

  \vspace{0.2em}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item $\bar{A} \in \R^{N \times N}$: state transition matrix
        \item $\bar{B} \in \R^{N \times 1}$: input projection
        \item $C \in \R^{1 \times N}$: output projection
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{block}{Linear Time-Invariance (LTI)}
        $(\bar{A}, \bar{B}, C)$ are \emph{constant}: they do not depend on the input.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Discretization}

  \[
    A = \exp(\Delta \bar{A}), \qquad B = (\Delta \bar{A})^{-1}(\exp(\Delta \bar{A}) - I) \cdot \Delta \bar{B}
  \]

\begin{frame}{The Dual View: Recurrence vs.\ Convolution}

  For LTI systems ($A, B, C$ constant), unrolling the recurrence yields:
  \[
    y_t = C A^t B\, x_0 + C A^{t-1} B\, x_1 + \cdots + C B\, x_t = \sum_{k=0}^{t} \underbrace{C A^k B}_{\bar{K}_k}\, x_{t-k}
  \]

  This is a \textbf{causal convolution} $y = \bar{K} * x$ with kernel $\bar{K} \in \R^T$.

  \vspace{0.5em}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Recurrent mode}
      \begin{itemize}
        \item $O(N)$ per step
        \item Sequential: good for inference
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Convolutional mode}
      \begin{itemize}
        \item Compute $\bar{K}$, then use FFT
        \item $O(T \log T)$: good for training
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.5em}
  \begin{block}{S4~\cite{gu2022s4}}
    Structured State Spaces for Sequences (S4) uses HiPPO initialization and diagonal structure for $\bar{A}$ to make this practical. First SSM to match Transformers on long-range benchmarks.
  \end{block}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{The Limitation of LTI Models}

  LTI SSMs apply the \emph{same} convolution kernel to every input, regardless of content.

  \vspace{0.5em}
  \textbf{Example:} Consider the task of selectively copying tokens from context.

  \begin{center}
    \begin{tikzpicture}[
      tok/.style={minimum width=0.65cm, minimum height=0.55cm, draw, font=\small, inner sep=1pt},
    ]
      \node[tok, fill=mambablue!20] at (0,0) {A};
      \node[tok] at (0.7,0) {x};
      \node[tok] at (1.4,0) {x};
      \node[tok, fill=mambared!20] at (2.1,0) {B};
      \node[tok] at (2.8,0) {x};
      \node[tok, fill=mambagreen!20] at (3.5,0) {C};
      \node[tok] at (4.2,0) {x};
      \node[tok] at (4.9,0) {x};

      \draw[-Stealth, thick] (5.4,0) -- (6.0,0);

      \node[tok, fill=mambablue!20] at (6.5,0) {A};
      \node[tok, fill=mambared!20] at (7.2,0) {B};
      \node[tok, fill=mambagreen!20] at (7.9,0) {C};
    \end{tikzpicture}
  \end{center}

  \begin{itemize}
    \item An LTI model treats every token identically. It cannot \emph{selectively} remember some tokens and ignore others based on content
    \item A fixed kernel is fundamentally \alert{content-agnostic}
    \item Transformers solve this via content-dependent attention weights
  \end{itemize}

  \vspace{0.3em}
  \begin{alertblock}{Insight}
    To make SSMs truly powerful, the parameters must be \textbf{input-dependent}.
    This breaks LTI and the convolutional mode, but enables \emph{selectivity}.
  \end{alertblock}
\end{frame}

\subsection{Mamba: Selective State Spaces}

% -----------------------------------------------------------------------
\begin{frame}{Selectivity: The Core Idea of Mamba~\cite{gu2024mamba}}

  \textbf{Mamba} makes the SSM parameters \emph{functions of the input}:
  \[
    B_t = \text{Linear}_B(x_t), \quad C_t = \text{Linear}_C(x_t), \quad \Delta_t = \text{softplus}(\text{Linear}_\Delta(x_t))
  \]

  The discrete recurrence becomes \textbf{time-varying}:
  \[
    \boxed{h_t = A_t\, h_{t-1} + B_t\, x_t, \qquad y_t = C_t^\top h_t}
  \]

  where $A_t = \exp(\Delta_t \cdot \bar{A})$ and $B_t, C_t$ depend on $x_t$.

  \vspace{0.5em}
  \textbf{Intuition for $\Delta_t$ (step size):}
  \begin{itemize}
    \item \textbf{Large $\Delta_t$}: Focus on current input, reset state $\approx$ ``remember this token''
    \item \textbf{Small $\Delta_t$}: Ignore current input, preserve state $\approx$ ``skip this token''
  \end{itemize}

  \vspace{0.3em}
  The model learns \emph{when} to update its state. This is the selection mechanism.
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{The Hardware Challenge}

  Making parameters input-dependent \alert{breaks} the convolutional mode:
  \begin{itemize}
    \item Time-varying $(A_t, B_t, C_t)$ $\Rightarrow$ no fixed convolution kernel
    \item The recurrence must be computed sequentially $\Rightarrow$ back to the RNN problem?
  \end{itemize}

  \vspace{0.5em}
  \textbf{The problem is not FLOPs; it's memory access.}

  Modern GPUs have:
  \begin{center}
    \begin{tabular}{lcc}
      \toprule
      & \textbf{SRAM (on-chip)} & \textbf{HBM (off-chip)} \\
      \midrule
      Size & $\sim$20 MB & $\sim$80 GB \\
      Bandwidth & $\sim$19 TB/s & $\sim$3 TB/s \\
      \bottomrule
    \end{tabular}
  \end{center}

  \vspace{0.3em}
  A naive scan reads/writes the full state from HBM at every step. Mamba's selective scan is \emph{memory-bound}, not compute-bound.

  \vspace{0.3em}
  \begin{block}{Solution: Hardware-aware selective scan}
    Fuse the entire recurrence into a single GPU kernel that keeps the state in \textbf{SRAM}, avoiding repeated HBM round-trips. Same idea as FlashAttention~\cite{dao2022flashattention}, applied to scans.
  \end{block}
\end{frame}

    \item Discretization + scan + output projection all happen \textbf{in SRAM}
    \item Intermediate states $h_t$ never written to HBM. Only the final output $y$
    \item Recomputation in backward pass instead of storing activations
  \end{itemize}
\end{frame}

\begin{frame}{Mamba-1 Architecture}

  \begin{columns}
    \begin{column}{0.55\textwidth}
      Each Mamba layer processes tokens via:
      \begin{enumerate}
        \item \textbf{Two parallel branches} from input projection:
          \begin{itemize}
            \item Branch A: Linear $\to$ Conv1d $\to$ SiLU $\to$ SSM
            \item Branch B: Linear $\to$ SiLU (gate)
          \end{itemize}
        \item SSM branch produces $B_t, C_t, \Delta_t$ from the activated input
        \item Selective scan computes output
        \item \textbf{Gated output}: multiply SSM output with Branch B, then project
      \end{enumerate}

      \vspace{0.5em}
      \textbf{Key design choices:}
      \begin{itemize}
        \item $A$ is diagonal with $N$ elements per channel
        \item $P = 1$: each channel has its own SSM (no shared heads)
        \item Typical: $N = 16$, $d_\text{model}$ up to 2048+
      \end{itemize}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{tikzpicture}[
        box/.style={draw, rounded corners, fill=mambablue!10, text width=2.8cm,
                    align=center, font=\small, minimum height=0.6cm},
        arr/.style={-Stealth, thick},

\begin{frame}{Mamba-1: Results}

  \textbf{Language modeling} (trained on the Pile, Chinchilla scaling~\cite{hoffmann2022chinchilla}):
  \begin{itemize}
    \item Matches Transformer quality at all scales (125M to 1.4B parameters)
    \item \textbf{5$\times$ higher throughput} than Transformer at inference (long sequences)
    \item Linear scaling in sequence length vs.\ quadratic for attention
  \end{itemize}

  \vspace{0.5em}
  \textbf{Long-range benchmarks:}
  \begin{itemize}
    \item State-of-the-art on Long Range Arena, speech, genomics
    \item Scales to sequences of 1M+ tokens at inference
  \end{itemize}

  \vspace{0.5em}
  \begin{alertblock}{Remaining issues}
    \begin{itemize}
      \item Training uses a \textbf{sequential scan}, no matrix multiplications
      \item GPUs/TPUs are optimized for matmuls, not sequential operations
      \item Despite kernel fusion, training throughput is \alert{lower than Transformers}
      \item Each channel has an independent SSM ($P = 1$), limited parameter sharing
    \end{itemize}
  \end{alertblock}
\end{frame}

% =======================================================================
\subsection{Mamba-2: State Space Duality}

% -----------------------------------------------------------------------
\begin{frame}{Two Problems with Mamba-1}

  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Problem 1: Understanding}
      \begin{itemize}
        \item State space models and attention mechanisms look very different
        \item Are they fundamentally different, or related?
        \item Can we unify them in a single framework?
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Problem 2: Efficiency}
      \begin{itemize}
        \item Despite optimization efforts, Mamba-1 training remains \alert{less hardware-efficient} than attention
        \item SSMs lack native support for matrix multiplications
        \item GPUs/TPUs are built to do matmuls fast
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{1em}
  \begin{alertblock}{The SSD Framework~\cite{dao2024mamba2}}
    Both problems are solved together: by revealing that a scalar-structured SSM \emph{is} a form of attention, we unlock matmul-based hardware-efficient training.
  \end{alertblock}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{Restricting $A$: Scalar-Times-Identity}

  \textbf{The one structural change in Mamba-2:}

  \begin{center}
    \begin{tabular}{lcl}
      \toprule
      Structure & Shape of $A$ & Description \\
      \midrule
      General   & $(T, N, N)$   & Unrestricted matrix \\
      Diagonal  & $(T, N)$      & Mamba-1 \\
      \textbf{Scalar $\times$ Identity} & $(T)$ & \textbf{Mamba-2 / SSD} \\
      \bottomrule
    \end{tabular}
  \end{center}

  \vspace{0.5em}
  By forcing all diagonal elements of $A_t$ to be the \emph{same} scalar $a_t$:
  \[
    A_t = a_t \cdot I_N, \qquad a_t \in \R
  \]

  \begin{itemize}
    \item Fewer parameters per step, but enables a fundamentally new algorithm
    \item Opens the door to the \textbf{quadratic (attention) mode}
  \end{itemize}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{Multihead SSMs}

  For real language models, inputs have $\mathtt{P}$ channels: $X \in \R^{(T, P)}$

  \[
    Y^{(T,P)} = \mathsf{SSM}\!\left(A^{(T)},\, B^{(T,N)},\, C^{(T,N)}\right)\!\left(X^{(T,P)}\right)
  \]

  \vspace{0.4em}
  \textbf{Multihead structure} (like Transformers):
  \begin{itemize}
    \item Heads of dimension $\mathtt{P} \in \{64, 128\}$ (vs.\ $\mathtt{P}=1$ in Mamba-1)
    \item Within a head, the \emph{same} $(a_t, B_t, C_t)$ are applied to all $\mathtt{P}$ channels
    \item Across heads, $(a, B, C)$ are independent
  \end{itemize}

  \vspace{0.5em}
  \begin{block}{Key advantage over Mamba-1}
    Larger state dimensions become tractable: $\mathtt{N} = 16$ (Mamba-1) $\to$ $\mathtt{N} \in [64, 256]$ (Mamba-2), while \emph{simultaneously} training faster.
  \end{block}

\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{Unrolling the SSM: The Attention Mode}

  With scalar $A_t = a_t$, unroll the recurrence over the full sequence. Define:
  \[
    a_{i:j}^\times \;=\; a_i \cdot a_{i-1} \cdots a_{j+1}
    \qquad \text{(cumulative product of decays)}
  \]

  The output at time $i$ becomes:
  \[
    y_i = \sum_{j \le i} C_i^\top \left(a_{i:j}^\times\right) B_j\, x_j
  \]

  Stacking all $y_i$ gives a matrix equation $Y = M X$ where:
  \[
    M = L \circ C B^\top \in \R^{(T,T)}
  \]

  $L$ is the lower-triangular \textbf{decay matrix}:
  \[
    L_{ij} = \begin{cases} a_{i:j}^\times & i \ge j \\ 0 & i < j \end{cases}
  \]
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{The Decay Matrix $L$}

  \[
    L = \begin{bmatrix}
      1 & & & & \\
      a_1 & 1 & & & \\
      a_2 a_1 & a_2 & 1 & & \\
      \vdots & \vdots & \ddots & \ddots & \\
      a_{T-1}\!\cdots\!a_1 & a_{T-1}\!\cdots\!a_2 & \cdots & a_{T-1} & 1
    \end{bmatrix}
  \]

  \vspace{0.5em}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Special case:} $a_t = 1$ for all $t$:
      \[
        L = \text{lower-triangular causal mask}
      \]
      and the model becomes \emph{causal linear attention}:
      \[
        Y = (L \circ QK^\top)\, V
      \]
      where $(C, B, X) \leftrightarrow (Q, K, V)$.
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{alertblock}{Interpretation of $a_t$}
        The scalar decays $a_t \in (0,1)$ act as \textbf{input-dependent relative positional encodings}: they control how much past information is retained at each step.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{State Space Duality: Same Model, Two Views}

  \begin{center}
    \large
    $(A^{(T)},\; B^{(T,N)},\; C^{(T,N)},\; X^{(T,P)}) \;\mapsto\; Y^{(T,P)}$
  \end{center}

  \vspace{0.5em}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{SSM (Linear) Mode}
      \[h_t = a_t h_{t-1} + B_t x_t,\quad y_t = C_t^\top h_t\]
      \begin{itemize}
        \item Constant-size state $\mathtt{N}$
        \item $O(T)$ computation
        \item No matmuls $\to$ slow on hardware
        \item Efficient at \emph{inference}
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Attention (Quadratic) Mode}
      \[Y = (L \circ CB^\top)\, X\]
      \begin{itemize}
        \item Materializes $T \times T$ matrix
        \item $O(T^2 N)$ computation
        \item Pure matmuls $\to$ fast on hardware
        \item Efficient at \emph{training} (small $T$)
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.5em}
  \begin{block}{The duality}
    These are \textbf{mathematically identical}. The same model can be computed either way.
  \end{block}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{SSD vs.\ Standard Attention: Two Differences}

  \begin{center}
  \begin{tabular}{lcc}
    \toprule
    & \textbf{Standard Attention} & \textbf{SSD} \\
    \midrule
    Sequence mixing matrix & $(L_{\text{causal}} \circ \text{softmax}(QK^\top))\,V$ & $(L \circ CB^\top)\,X$ \\
    Softmax & \checkmark & \texttimes \\
    Multiplicative mask $L$ & causal only ($a_t=1$) & \checkmark (input-dependent $a_t$) \\
    \midrule
    State size at inference & $O(T)$ (KV cache) & $O(N)$ \textbf{(constant!)} \\
    Training complexity & $O(T^2 N)$ & $O(T N^2)$ \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{itemize}
    \item Removing softmax: state stays \emph{constant} size, inference is $O(1)$ per step
    \item The decay mask $L$ is what distinguishes SSD from linear attention and encodes \emph{selectivity}
  \end{itemize}
\end{frame}

% -----------------------------------------------------------------------
\begin{frame}{Efficiency Comparison}

  \begin{center}
  \begin{tabular}{lcccc}
    \toprule
    & \textbf{Attention} & \textbf{SSM} & \textbf{SSD} \\
    \midrule
    State size      & $T$           & $N$         & $\bm{N}$    \\
    Training FLOPs  & $T^2 N$       & $TN^2$      & $\bm{TN^2}$ \\
    Inference FLOPs & $TN$          & $N^2$       & $\bm{N^2}$  \\
    Naive memory    & $T^2$         & $TN^2$      & $\bm{TN}$   \\
    Matrix multiplications & \checkmark & \texttimes & \checkmark \\
    \bottomrule
  \end{tabular}
  \end{center}

  \vspace{0.5em}
  \begin{alertblock}{The best of both worlds}
    SSD achieves \textbf{constant inference state} (like SSMs) and \textbf{matrix multiply-based training} (like attention) simultaneously.
  \end{alertblock}
\end{frame}

\subsection{The Efficient SSD Algorithm}

% -----------------------------------------------------------------------
\begin{frame}{The Chunked SSD Algorithm}

  Neither pure mode is ideal for training on long sequences. The solution: \textbf{chunk the sequence}.

  \vspace{0.5em}
  Split $T$ tokens into $Q = T / C$ chunks of size $C$:

  \begin{enumerate}
    \item \textbf{Intra-chunk} (quadratic/attention mode):
      \begin{itemize}
        \item Within each chunk, compute $Y_{\text{intra}} = (L \circ CB^\top) X$ via matmul
        \item Cost: $O(C^2 \cdot P)$ per chunk, parallelizable across batch and heads
      \end{itemize}
    \item \textbf{Inter-chunk} (SSM/linear mode):
      \begin{itemize}
        \item Compute a \emph{summary state} $h$ at the end of each chunk
        \item Pass states across chunk boundaries sequentially ($Q$ steps, not $T$)
        \item For each chunk, add the contribution of all past states: $Y_{\text{inter}}$
      \end{itemize}
    \item \textbf{Combine:} $Y = Y_{\text{intra}} + Y_{\text{inter}} + D \odot X$
  \end{enumerate}

  \vspace{0.3em}
  Overall cost: $O\!\left(\frac{T}{C} \cdot C^2 N\right) = O(TCN)$, optimal for $C = \sqrt{T}$.
\end{frame}

\subsection{The Mamba-2 Architecture}

% -----------------------------------------------------------------------
\begin{frame}{Mamba-2 Architecture Overview}

  \begin{columns}
    \begin{column}{0.55\textwidth}
      \textbf{Key architectural change from Mamba-1:}
      \vspace{0.3em}

      $(A, B, C)$ are produced \emph{in parallel} with input $X$, not sequentially.

      \vspace{0.8em}
      \textbf{Per-layer computation:}
      \begin{enumerate}
        \item Linear projection: $[Z, X, B, C, \Delta] = \text{Linear}(u)$
        \item Conv1d (causal, depthwise) on $X$
        \item $A \leftarrow -\exp(\texttt{A\_log})$, $\Delta \leftarrow \text{softplus}(\Delta + \texttt{dt\_bias})$
        \item SSD block: $Y = \text{SSD}(A \cdot \Delta,\; B,\; C,\; X \cdot \Delta)$
        \item Output: $\text{out\_proj}(\text{RMSNorm}(Y) \cdot \text{SiLU}(Z))$
      \end{enumerate}
    \end{column}

\begin{frame}{Summary: State Space Models and Mamba}

  \begin{enumerate}
    \item \textbf{State Space Models}: Linear systems discretized from continuous time; dual recurrent/convolutional view enables parallel training for LTI models
    \item \textbf{Mamba-1}: Makes SSM parameters \emph{input-dependent} (selective); hardware-aware scan keeps computation in SRAM; matches Transformer quality with linear-time inference
    \item \textbf{State Space Duality}: Scalar-structured SSMs and causal linear attention are the \emph{same} model viewed differently
    \item \textbf{Mamba-2}: Chunked SSD algorithm: $O(TCN)$ compute, constant inference state, full matmul utilization; 2--8$\times$ faster training than Mamba-1
  \end{enumerate}
